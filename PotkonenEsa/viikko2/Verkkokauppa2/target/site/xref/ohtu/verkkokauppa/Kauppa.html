13      private IViitegeneraattori viitegeneraattori;

37      public void poistaKorista(int id) {

40      }
41  
42      /* (non-Javadoc)
43  	 * @see ohtu.verkkokauppa.IKauppa#lisaaKoriin(int)
44  	 */
45      public void lisaaKoriin(int id) {
46          if (varasto.saldo(id)>0) {
47              Tuote t = varasto.haeTuote(id);             
48              ostoskori.lisaa(t);
49              varasto.otaVarastosta(t);
50          }
51      }
52

55  	 */
56      public boolean tilimaksu(String nimi, String tiliNumero) {
57          int viite = viitegeneraattori.uusi();
58          int summa = ostoskori.hinta();
59          
60          return pankki.tilisiirto(nimi, viite, tiliNumero, kaupanTili, summa);
61      }
